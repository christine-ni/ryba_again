name: learn_workflow   	#название воркфлоу, которое будет отображаться в окошке "Actions" на  гитхабе
run-name: ${{ github.actor }} is learning Workflows 	#указывает имя конкретного запуска процесса
on: [push] 	#определяем, при каком событии запускается воркфлоуу
jobs:	#определяем набор задач, которые должны быть выполнены в рамках воркфлоу
  check-bats-version:	#название задачи. Это имя используется для ссылки на задачу в других частях конфигурации или при отображении в интерфейсе.
    runs-on: ubuntu-latest	#операционная система,  на которой будет выполняться задача
    steps:	#список шагов,  которые должны быть выполнены в задаче
      - uses: actions/checkout@v4	# используемое действие GitHub Actions. Здесь происходит клонирование репозитория в рабочую область.
      - uses: actions/setup-node@v4	#настройка среды работы с Node.js
        with:	#задаем параметры для действия
          node-version: '20'	#версия Node.js
      - run: npm install -g bats	#выполнение команды в командной строке.  Здесь NPM устанавливает инструмент для тестирования  bats
      - run: bats -v	#вывод текущей версии bats
